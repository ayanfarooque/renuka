#include<stdio.h>
#include<string.h>
// Function to find gcd
int gcd(int a, int b)
{
    while(b != 0)
    {
        int t = b;
        b = a % b;
        a = t;
    }
    return a;
}

// Function to find modular inverse (d)
int modInverse(int e, int phi)
{
    for(int d = 1; d < phi; d++)
    {
        if((d * e) % phi == 1)
            return d;
    }
    return -1;
}

int main()
{
    int p, q, n, phi, e, d;
    int k, b, bin[20], i;

    char text[200];
    long int enc[200];
    int size;

    // Input primes
    printf("Enter two prime numbers (p and q): ");
    scanf("%d %d", &p, &q);

    n = p * q;
    phi = (p - 1) * (q - 1);

    // Find e automatically
    for(e = 2; e < phi; e++)
    {
        if(gcd(e, phi) == 1)
            break;
    }

    // Find d
    d = modInverse(e, phi);

    printf("\nComputed values:\n");
    printf("n = %d\n", n);
    printf("phi = %d\n", phi);
    printf("e = %d\n", e);
    printf("d = %d\n", d);

    getchar(); // clear buffer

    // Input text
    printf("\nEnter text: ");
    fgets(text, sizeof(text), stdin);
    text[strcspn(text, "\n")] = 0;

    // ---------- ENCRYPTION ----------
    i = -1;
    b = e;
    while(b > 0)
    {
        bin[++i] = b % 2;
        b = b / 2;
    }
    int k1 = i;

    printf("\nEncrypted values:\n");

    size = strlen(text);

    for(int j = 0; j < size; j++)
    {
        long int m = text[j];
        long int temp = 1;

        for(i = k1; i >= 0; i--)
        {
            temp = (temp * temp) % n;
            if(bin[i] == 1)
                temp = (temp * m) % n;
        }

        enc[j] = temp;
        printf("%ld ", enc[j]);
    }

    // ---------- DECRYPTION ----------
    i = -1;
    b = d;
    while(b > 0)
    {
        bin[++i] = b % 2;
        b = b / 2;
    }
    int k2 = i;

    printf("\n\nDecrypted text:\n");

    for(int j = 0; j < size; j++)
    {
        long int temp = 1;
        long int c = enc[j];

        for(i = k2; i >= 0; i--)
        {
            temp = (temp * temp) % n;
            if(bin[i] == 1)
                temp = (temp * c) % n;
        }

        printf("%c", (char)temp);
    }

    printf("\n");

    return 0;
}
